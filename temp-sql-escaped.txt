DEFAULT '[]',\n  image_url TEXT,\n  stripe_price_id TEXT,\n  download_url TEXT,\n  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()\n);\n\n-- Orders table\nCREATE TABLE IF NOT EXISTS orders (\n  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),\n  stripe_session_id TEXT UNIQUE NOT NULL,\n  customer_email TEXT NOT NULL,\n  customer_name TEXT,\n  total_cents INTEGER NOT NULL,\n  status TEXT NOT NULL DEFAULT 'pending',\n  created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()\n);\n\n-- Order items table\nCREATE TABLE IF NOT EXISTS order_items (\n  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),\n  order_id TIMESTAMPTZ NOT NULL DEFAULT NOW()\n);\n\n-- Downloads table\nCREATE TABLE IF NOT EXISTS downloads (\n  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),\n  order_id UUID NOT NULL REFERENCES orders(id) read access\nCREATE POLICY \"Products are viewable by everyone\"\n  ON products FOR SELECT\n  USING (true);\n\n-- Orders: readable by customer email match\nCREATE POLICY \"Orders are viewable by customer\"\n  ON orders FOR SELECT\n  USING (auth.jwt() ->> 'email' = customer_email);\n\n-- Order items: readable by order owner\nCREATE POLICY \"Order items viewable by order owner\"\n  ON order_items FOR SELECT\n  USING (\n    EXISTS (\n      SELECT 1 FROM orders\n      WHERE orders.id = order_items.order_id\n      AND orders.customer_email = auth.jwt() ->> 'email'\n    )\n  );\n\n-- Downloads: readable by token (no auth required for download links)\nCREATE POLICY \"Downloads viewable by token\"\n  ON downloads FOR SELECT\n  USING (true);\n\n-- Indexes for performance\nCREATE INDEX IF NOT EXISTS idx_orders_stripe_session ON orders(stripe_session_id);\nCREATE INDEX IF NOT EXISTS idx_orders_email ON orders(customer_email);\nCREATE INDEX IF NOT EXISTS idx_order_items_order ON order_items(order_id);\nCREATE INDEX IF NOT EXISTS idx_downloads_token ON downloads(download_token);\nCREATE INDEX IF NOT EXISTS idx_downloads_order ON downloads(order_id);\n\n-- Seed products from
